class ig:
    def __init__(self, follow, unfollow, execute, reg, tm):
        self.follow = follow
        self.unfollow = unfollow
        self.execute = execute
        self.reg = reg
        self.tm = tm

    def unfollowing(self):
        e = 0
        roll = 0
        while True:
            if exists(self.unfollow):

                click(self.unfollow)
                wait(self.execute, FOREVER)
                click(self.execute)
                e += 1
                if e == self.tm:
                    popup("berhasil execute {}".format(self.tm))
                    quit()
                wait(2)


            else:
                for x in range(4):
                    type(Key.DOWN *10)
                roll += 1
                if roll >= 5:
                    quit()

    def following(self):
        f = 0
        roll = 0
        while True:
            if exists(self.follow):

                click(self.follow)

                f += 1
                #reset roll
                roll = 0

                if f == self.tm:
                    popup("Berhasil action {}x".format(self.tm))
                    quit()

                wait(2)

            else:
                for z in range(4):

                    type(Key.DOWN)

                wait(1)
                roll += 1
                if roll >= 2:
                    popup("sudah tidak ada yang bisa di clik..!!")
                    quit()
